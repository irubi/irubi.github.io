#include "stdio.h" 
void main() 
{ 
  int n; //杨辉三角的阶数 
  int **c;    //存储杨辉三角的二维数组 
  int i,j; //二层循环变量 
  int k; 
  printf("Please input the layer of yanghui:"); 
  scanf("%d",&n); 
  c=new int* [n];    //动态分配二维空间 
        for(k=0;k<n;k++) 
    c[k]=new int[n]; 
 
  for(k=0;k<n;k++)            
     { 
    c[k][0]=1; 
         c[k][k]=1; 
     } 
  for(i=2;i<n;i++) 
     for(j=1;j<i;j++) 
    c[i][j]=c[i-1][j-1]+c[i-1][j]; 
    
 
        for(i=0;i<n;i++)                 //输出结果 
  { 
    for(j=0;j<=i;j++) 
    { 
      printf("%d ",c[i][j]); 
    } 
    printf("\n"); 
  } 
}
