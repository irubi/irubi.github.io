##3个月大开始生兔子是fib序列。
class Rabbit 
    @@count=0 
    attr_accessor :age 
    def initialize 
        @age=0 
        @@count+=1 
    end 

    def increment 
        @age+=1 
    end 

    def self.quantity 
        @@count 
    end 
end 

first_rabbit=Rabbit.new 
rabbits=Array.new 
rabbits<<first_rabit
