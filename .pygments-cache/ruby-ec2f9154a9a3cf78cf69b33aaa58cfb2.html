def sub(n) 
    a=Array.new 
    flag=1 
    while flag==1 
        for i in(2..n) 
            if n%i==0 
                a<<i 
                n=n/i 
                break 
            end 
            if i==n-1 and n%i!=0 
                flag=0 
            end 
        end 
    end 
    a 
end 
puts "Input a number:" 
n=gets.chomp.to_i 
puts "#{n}=#{sub(n).join('*')}"
