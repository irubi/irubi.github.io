def gu(n) 
    if n==0 
        print "Invalid number!" 
    end 
    while n!=1 
        if n%2==0 
            yield(n) 
            n=n/2 
        elsif n%2!=0 
            yield(n) 
            n=n*3+1 
        end 
    end 
    print n 
end
