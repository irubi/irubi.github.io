def yang_triangle(n) 
    a=[] 
    for f in (0..n-1) 
        a[f]=[] 
        for n in (0..f+1) 
            a[f][0]=1 
            if f>1 
                for m in (1..n-2) 
                    a[f][m]=a[f-1][m-1]+a[f-1][m] 
                end 
            end 
            a[f][n-1]=1 
        end 
    end 
    a 
end 
puts "Input a number:" 
n=gets.chomp.to_i 
yang_triangle(n).each do |arr| 
    arr.each do |elem| 
        print elem, " " 
    end 
end
