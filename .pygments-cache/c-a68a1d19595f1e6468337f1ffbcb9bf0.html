#include "stdio.h" 
#include "math.h" 
#define MAX 1000 
void main() 
{ 
  int temp[MAX/2]; 
  int sum=0; 
  int k; 
  for(int m=0;m<MAX/2;m++) //初始化 
    temp[m]=0; 
  for(int i=2;i<MAX;i++) 
  { 
    k=0; 
    for(int j=1;j<=floor(i/2);j++) 
    { 
      if(i%j==0) 
      { 
        temp[k]=j; 
        sum+=j; 
        k++; 
      }//if 
        
    } 
 
     
    if(sum==i) 
    { 
      printf("\n%d=",i); 
      for(k=0;k<MAX/2;k++) 
      {                
        printf("%d",temp[k]); 
        if(temp[k+1]==0) 
          break; 
        printf("+"); 
      }//for 
      sum=0; 
      for(int m=0;m<MAX/2;m++)    
                                temp[m]=0;    
    }//if 
    else    
    { 
      sum=0;                         
    } 
  }//for 
}//main
